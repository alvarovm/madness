43 class BS_MomentFunctor : public FunctionFunctorInterface<double, 3> {
44  private:
45  const int i, j, k;
46 
47  public:
48  BS_MomentFunctor(int i, int j, int k) : i(i), j(j), k(k) {}
49  explicit BS_MomentFunctor(const std::vector<int>& x) : i(x[0]), j(x[1]), k(x[2]) {}
50  double operator()(const Vector<double, 3>& r) const {
51  double xi = 1.0, yj = 1.0, zk = 1.0;
52  for (int p = 0; p < i; ++p) xi *= r[0];
53  for (int p = 0; p < j; ++p) yj *= r[1];
54  for (int p = 0; p < k; ++p) zk *= r[2];
55  return xi * yj * zk;
56  }

61 /// the function looks like
62 /// \[

65 template <std::size_t NDIM>
66 class GaussianGuess : public FunctionFunctorInterface<double, NDIM> {
67  typedef Vector<double, NDIM> coordT;
68 
69  public:
70  /// ctor
71 
72  /// @param[in] origin the origin of the Gauss function
73  /// @param[in] alpha the exponent exp(-alpha r^2)
74  /// @param[in] ijk the monomial x^i y^j z^k exp(-alpha r^2) (for NDIM)
75  GaussianGuess(const coordT& origin, const double alpha, const std::vector<int> ijk = std::vector<int>(NDIM))
76  : origin(origin), exponent(alpha), ijk(ijk) {}
77 
78  coordT origin;
79  double exponent; ///< exponent of the guess
80  std::vector<int> ijk; ///< cartesian exponents
81 
82  double operator()(const coordT& xyz) const {
83  double arg = 0.0, prefac = 1.0;
84  for (std::size_t i = 0; i < NDIM; ++i) {
85  arg += (xyz[i] - origin[i]) * (xyz[i] - origin[i]);
86  prefac *= pow(xyz[i], ijk[i]);
87  }
88  const double e = exponent * arg;
89  return prefac * exp(-e);
90  }
91 };
