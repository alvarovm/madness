43 /// Disables default copy constructor and assignment operators
44 
45 /// From http://home.pb.net/~tglenn/CPPNOTES.html. Inherit from this
46 /// class in order to inhibit the automatic generation of the default
47 /// copy constructor and the default assignment operator of the
48 /// derived class.

52  NO_DEFAULTS(const NO_DEFAULTS&) = delete;
53  NO_DEFAULTS(NO_DEFAULTS&&) = delete;
