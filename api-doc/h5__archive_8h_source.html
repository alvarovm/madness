159  void open(const char* filename) {
160 
161  file_id = H5Fopen(filename, H5F_ACC_RDONLY, H5P_DEFAULT);
162 //
163  dataset_id = H5Dopen2(file_id, "/dset", H5P_DEFAULT);
164 //
165  hid_t datatype = H5Dget_type(dataset_id);
166  size_t size = H5Tget_size(datatype);
167  H5T_class_t t_class = H5Tget_class(datatype);
168 
169  hid_t dataspace_id = H5Dget_space(dataset_id); /* dataspace handle */
170  int rank = H5Sget_simple_extent_ndims(dataspace_id);
171  hsize_t dims_out[1];
172  status = H5Sget_simple_extent_dims(dataspace_id, dims_out, NULL);
173  v.resize((unsigned long)(dims_out[0]));
174  status = H5Dread(dataset_id, H5T_NATIVE_UCHAR, H5S_ALL, H5S_ALL, H5P_DEFAULT, v.data());

183  load(T* t, long n) const {
184  var.load(t,n);
