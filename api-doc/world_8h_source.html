332  /// Associate a local pointer with a universe-wide unique ID.
333 
334  /// Use the routines \c register_ptr(), \c unregister_ptr(),
335  /// \c id_from_ptr(), \c ptr_from_id() to map distributed data
336  /// structures identified by the unique ID to/from
337  /// process-local data.
338  ///
339  /// \note The pointer will be internally cast to a (void *),
340  /// so don't use member pointers here.
341  ///
342  /// \note All unique objects of any type within a \c World must
343  /// presently be created in the same order on all processes so
344  /// as to provide the uniquess property without global
345  /// communication.
346  /// \tparam T The type of data to be associated.
347  /// \param[in] ptr Pointer to the data that will be associated
348  /// with a unique ID.
349  /// \return The unique ID associated with the supplied data.
350  template <typename T>
351  uniqueidT register_ptr(T* ptr) {
352  MADNESS_ASSERT(sizeof(T*) == sizeof(void *));
353  uniqueidT id = unique_obj_id();
354  map_id_to_ptr.insert(std::pair<uniqueidT,void*>(id,static_cast<void*>(ptr)));
355  map_ptr_to_id.insert(std::pair<void*,uniqueidT>(static_cast<void*>(ptr),id));
356  return id;
357  }
