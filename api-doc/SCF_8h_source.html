105 inline double mask1(double x) {
106  /* Iterated first beta function to switch smoothly
107  from 0->1 in [0,1]. n iterations produce 2*n-1
108  zero derivatives at the end points. Order of polyn
109  is 3^n.
110 
111  Currently use one iteration so that first deriv.
112  is zero at interior boundary and is exactly representable
113  by low order multiwavelet without refinement */
114 
115  x = (x * x * (3. - 2. * x));
116  return x;
117 }

119 static double mask3(const coordT& ruser) {
120  coordT rsim;
121  user_to_sim(ruser, rsim);
122  double x = rsim[0], y = rsim[1], z = rsim[2];
123  double lo = 0.0625, hi = 1.0 - lo, result = 1.0;
124  double rlo = 1.0 / lo;
125 
126  if (x < lo)
127  result *= mask1(x * rlo);
128  else if (x > hi)
129  result *= mask1((1.0 - x) * rlo);
130  if (y < lo)
131  result *= mask1(y * rlo);
132  else if (y > hi)
133  result *= mask1((1.0 - y) * rlo);
134  if (z < lo)
135  result *= mask1(z * rlo);
136  else if (z > hi)
137  result *= mask1((1.0 - z) * rlo);
138 
139  return result;
140 }

156 class MomentFunctor : public FunctionFunctorInterface<double, 3> {
157 private:

165  double xi = 1.0, yj = 1.0, zk = 1.0;
166  for (int p = 0; p < i; ++p) xi *= r[0];
167  for (int p = 0; p < j; ++p) yj *= r[1];
