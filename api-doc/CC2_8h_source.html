185  bool
186  /// Iterates the singles equations for CCS, CC2, LRCC2
187  /// The corresponding regulairzation tails of the doubles are updated in every iteration (therefore doubles are not marked as const)
188  /// @param[in] : singles, the singles that are iterated
189  /// @param[in] : singles2, additional set of singles for potential (in LRCC2 this are the Ground State singles)
190  /// @param[in] : gs_doubles, Ground State doubles (Needed for CC2 and LRCC2)
191  /// @param[in] : ex_doubles, Excited State doubles (Needed for LRCC2)
192  /// @param[in] : ctype: the calculation type: CCS, CC2, CC2_response_
193  /// @param[in] : maxiter: maxmial number of iterations
194  /// @param[out]: true if the overall change of the singles is below 10*donv_6D
195  iterate_singles(CC_vecfunction& singles, const CC_vecfunction singles2, Pairs<CCPair>& gs_doubles,
196  Pairs<CCPair>& ex_doubles, const CalcType ctype, const std::size_t maxiter) {
197  output.subsection("Iterate " + assign_name(ctype) + "-Singles");
198  CCTimer time_all(world, "Overall Iteration of " + assign_name(ctype) + "-Singles");
199  bool converged = true;
200 
201  CC_vecfunction old_singles(singles);
202  for (auto& tmp : singles.functions)
203  old_singles(tmp.first).function = copy(tmp.second.function);
204 
205  // KAIN solver
206  typedef allocator<double, 3> allocT;
207  typedef XNonlinearSolver<std::vector<Function<double, 3> >, double, allocT> solverT;
208  allocT alloc(world, singles.size());
209  solverT solver(allocT(world, singles.size()));
210  solver.do_print = (world.rank() == 0);
211 
212  for (size_t iter = 0; iter < maxiter; iter++) {
213  output.subsection("Microiteration " + std::to_string(iter) + " of " + assign_name(ctype) + "-Singles");
214  CCTimer time(world, "Microiteration " + std::to_string(iter) + " of " + assign_name(ctype) + "-Singles");
215  double omega = 0.0;
216  if (ctype == CT_LRCC2) omega = singles.omega;
217  else if (ctype == CT_LRCCS) omega = singles.omega;
218  else if (ctype == CT_ADC2) omega = singles.omega;
219 
220  // consistency check
221  switch (ctype) {
222  case CT_CC2:
223  if (singles.type != PARTICLE)
224  output.warning("iterate_singles: CC2 demanded but singles are not of type PARTICLE");
225  break;
226  case CT_MP2: MADNESS_EXCEPTION("Demanded Singles Calculation for MP2 ????", 1);
227  break;
228  case CT_LRCC2:
229  if (singles.type != RESPONSE or singles2.type != PARTICLE)
230  output.warning("iterate_singles: CC2_response_ singles have wrong types");
231  break;
232  case CT_LRCCS:
233  if (singles.type != RESPONSE)
234  output.warning("iterate_singles: CCS_response_ singles have wrong types");
235  break;
236  case CT_CISPD: MADNESS_EXCEPTION("Demanded Singles Calculation for CIS(D)", 1);
237  break;
238  case CT_ADC2:
239  MADNESS_ASSERT(singles.type == RESPONSE);
240  break;
241  case CT_TEST: MADNESS_EXCEPTION("Iterate Singles not implemented for Experimental calculation", 1);
242  break;
243  default: MADNESS_EXCEPTION(
244  ("Unknown calculation type in iterate singles: " + assign_name(ctype)).c_str(), 1);
245  }
246 
247  // get potentials
248  CCTimer time_V(world, assign_name(ctype) + "-Singles Potential");
249  vector_real_function_3d V;
250  if (ctype == CT_CC2) V = CCOPS.get_CC2_singles_potential_gs(singles, gs_doubles);
251  else if (ctype == CT_LRCC2)
252  V = CCOPS.get_CC2_singles_potential_ex(singles2, gs_doubles, singles, ex_doubles);
253  else if (ctype == CT_LRCCS) V = CCOPS.get_CCS_potential_ex(singles);
254  else if (ctype == CT_ADC2) V = CCOPS.get_ADC2_singles_potential(gs_doubles, singles, ex_doubles);
255  else MADNESS_EXCEPTION("iterate singles: unknown type", 1);
256  time_V.info(true, norm2(world, V));
257 
258  if (ctype == CT_LRCCS or ctype == CT_LRCC2 or ctype == CT_ADC2) {
259  omega = singles.omega; // computed with the potential
260  }
261 
262  scale(world, V, -2.0);
263  truncate(world, V);
264 
265  // make bsh operators
266  CCTimer time_makebsh(world, "Make G-Operators");
267  std::vector<std::shared_ptr<SeparatedConvolution<double, 3> > > G(singles.size());
268  for (size_t i = 0; i < G.size(); i++) {
269  const double bsh_eps = CCOPS.get_orbital_energies()[i + parameters.freeze()] + omega;
270  G[i] = std::shared_ptr<SeparatedConvolution<double, 3> >(
271  BSHOperatorPtr3D(world, sqrt(-2.0 * bsh_eps), parameters.lo(), parameters.thresh_bsh_3D()));
272  }
273  world.gop.fence();
274  time_makebsh.info();
275 
276  // apply bsh operators
277  CCTimer time_applyG(world, "Apply G-Operators");
278  vector_real_function_3d GV = apply<SeparatedConvolution<double, 3>, double, 3>(world, G, V);
279  world.gop.fence();
280  time_applyG.info();
281 
282  // apply Q-Projector to result
283  GV = CCOPS.apply_Qt(GV, CCOPS.mo_ket());
284 
285  // Normalize Singles if it is excited state
286  if (ctype == CT_LRCCS or ctype == CT_LRCC2 or ctype == CT_ADC2) {
287  output("Normalizing new singles");
288  const vector_real_function_3d x = GV;
289  const vector_real_function_3d xbra = mul(world, nemo->ncf->square(), GV);
290  const double norm = sqrt(inner(world, xbra, x).sum());
291  if (world.rank() == 0)
292  std::cout << " Norm was " << std::fixed << std::setprecision(parameters.output_prec()) << norm
293  << "\n";
294  scale(world, GV, 1.0 / norm);
295  } else output("Singles not normalized");
296 
297  // residual
298  const vector_real_function_3d residual = sub(world, singles.get_vecfunction(), GV);
299 
300  // information
301  const Tensor<double> R2xinnerx = inner(world, mul(world, nemo->ncf->square(), singles.get_vecfunction()),
302  singles.get_vecfunction());
303  const Tensor<double> R2GVinnerGV = inner(world, mul(world, nemo->ncf->square(), GV), GV);
304  const Tensor<double> R2rinnerr = inner(world, mul(world, nemo->ncf->square(), residual), residual);
305  const double R2vector_error = sqrt(R2rinnerr.sum());
306 
307  // print information
308  if (world.rank() == 0) std::cout << "\n\n-----Results of current interation:-----\n";
309  if (world.rank() == 0)
310  std::cout << "\nName: ||" << singles.name() << "||, ||GV" << singles.name() << ", ||residual||" << "\n";
311  if (world.rank() == 0)
312  std::cout << singles.name() << ": " << std::scientific << std::setprecision(parameters.output_prec())
313  << sqrt(R2xinnerx.sum()) << ", " << sqrt(R2GVinnerGV.sum()) << ", " << sqrt(R2rinnerr.sum())
314  << "\n----------------------------------------\n";
315  for (size_t i = 0; i < GV.size(); i++) {
316  if (world.rank() == 0)
317  std::cout << singles(i + parameters.freeze()).name() << ": " << std::scientific
318  << std::setprecision(parameters.output_prec())
319  << sqrt(R2xinnerx(i)) << ", " << sqrt(R2GVinnerGV(i)) << ", " << sqrt(R2rinnerr(i))
320  << "\n";
321  }
322  if (world.rank() == 0) std::cout << "\n----------------------------------------\n\n";
323 
324  // make second order update (only for response)
325  if (ctype == CT_LRCC2 or ctype == CT_LRCCS) {
326  output("\nMake 2nd order energy update:");
327  // include nuclear factors
328  {
329  vector_real_function_3d bra_res = mul(world, nemo->ncf->square(), residual);
330  vector_real_function_3d bra_GV = mul(world, nemo->ncf->square(), GV);
331  double Rtmp = inner(world, bra_res, V).sum();
332  double Rtmp2 = inner(world, bra_GV, GV).sum();
333  const double Rdelta = (0.5 * Rtmp / Rtmp2);
334  double old_omega = omega;
335  output("Delta-Update is not used");
336  if (world.rank() == 0)
337  std::cout << "omega, old_omega, delta" << std::fixed
338  << std::setprecision(parameters.output_prec() + 2) << omega << ", " << old_omega << ", "
339  << Rdelta << "\n\n";
340  }
341 
342  }
343 
344  // update singles
345  singles.omega = omega;
346  vector_real_function_3d new_singles = GV;
347  if (parameters.kain()) new_singles = solver.update(singles.get_vecfunction(), residual);
348  print_size(world, new_singles, "new_singles");
349  truncate(world, new_singles);
350  print_size(world, new_singles, "new_singles");
351  for (size_t i = 0; i < GV.size(); i++) {
352  singles(i + parameters.freeze()).function = copy(new_singles[i]);
353  }
354 
355  // update intermediates
356  CCOPS.update_intermediates(singles);
357 
358  // update reg_residues of doubles
359  //if(ctype==CC2_) update_reg_residues_gs(singles,gs_doubles);
360  //else if(ctype==LRCC2_) update_reg_residues_ex(singles2,singles,ex_doubles);
361 
362  converged = (R2vector_error < parameters.dconv_3D());
363 
364  time.info();
365  if (converged) break;
366  if (ctype == CT_LRCCS) break; // for CCS just one iteration to check convergence
367  }
368  time_all.info();
369 
370  // Assign the overall changes
371  bool no_change = true;
372  if (world.rank() == 0)
373  std::cout << "Change in Singles functions after all the CC2-Single-Microiterations" << std::endl;
374  for (auto& tmp : singles.functions) {
375  const double change = (tmp.second.function - old_singles(tmp.first).function).norm2();
376  tmp.second.current_error = change;
377  if (change > parameters.dconv_3D()) no_change = false;
378  if (world.rank() == 0)
379  std::cout << "Change of " << tmp.second.name() << "=" << tmp.second.current_error << std::endl;
380  }
381  // update reg_residues of doubles
382  if (ctype == CT_CC2) update_reg_residues_gs(singles, gs_doubles);
383  else if (ctype == CT_LRCC2) update_reg_residues_ex(singles2, singles, ex_doubles);
384 
385  //CCOPS.plot(singles);
386  singles.save_functions();
387  if (no_change) output("Change of Singles was below = " + std::to_string(parameters.dconv_3D()) + "!");
388  return no_change;
389  }
