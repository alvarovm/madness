65  explicit Atom(double x, double y, double z, double q, unsigned int atomic_number, bool pseudo_atom)
66  : x(x), y(y), z(z), q(q), atomic_number(atomic_number), pseudo_atom(pseudo_atom) {
67  mass= get_atomic_data(atomic_number).mass;
68 
69  if (mass==-1.0) MADNESS_EXCEPTION("faulty element in Atom",1);
70 
71  // unstable elements are indicated by negative masses, the mass
72  // is taken from the longest-living element

75  }

270  /// Apply to (x,y,z) a C2 rotation about an axis thru the origin and (xaxis,yaxis,zaxis)
271  struct apply_c2{
272  double xaxis, yaxis, zaxis;
273  apply_c2(double xaxis, double yaxis, double zaxis) : xaxis(xaxis), yaxis(yaxis), zaxis(zaxis) {}
274  void operator()(double& x, double& y, double& z) const {
275  double raxissq = xaxis*xaxis + yaxis*yaxis + zaxis*zaxis;
276  double dx = x*xaxis*xaxis/raxissq;
277  double dy = y*yaxis*yaxis/raxissq;
278  double dz = z*zaxis*zaxis/raxissq;
279  x = 2.0*dx - x;
280  y = 2.0*dy - y;
281  z = 2.0*dz - z;
282  }
283  };
284 
285  /// Apply to (x,y,z) a reflection through a plane containing the origin with normal (xaxis,yaxis,zaxis)
286  struct apply_sigma{
287  double xaxis, yaxis, zaxis;
288  apply_sigma(double xaxis, double yaxis, double zaxis) : xaxis(xaxis), yaxis(yaxis), zaxis(zaxis) {}
289  void operator()(double& x, double& y, double& z) const {
290  double raxissq = xaxis * xaxis + yaxis * yaxis + zaxis * zaxis;
291  double dx = x * xaxis * xaxis / raxissq;
292  double dy = y * yaxis * yaxis / raxissq;
293  double dz = z * zaxis * zaxis / raxissq;
294 
295  x = x - 2.0 * dx;
296  y = y - 2.0 * dy;
297  z = z - 2.0 * dz;
298  }
299  };
