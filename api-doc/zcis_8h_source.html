90  static std::vector<root> transform(World& world,
91  const std::vector<root>& v,
92  const Tensor<double_complex>& c,
93  bool fence=true) {
94 
95  int n = v.size(); // n is the old dimension
96  int m = c.dim(1); // m is the new dimension
97 
98  int nocca=v.front().afunction.size();
99  int noccb=v.front().bfunction.size();
100 
101  std::vector<root> result(m);
102 
103  for (auto&& r : result) {
104  r.afunction= zero_functions_compressed<double_complex,3>(world, nocca, false);
105  r.bfunction= zero_functions_compressed<double_complex,3>(world, noccb, false);
106  r.apot= zero_functions_compressed<double_complex,3>(world, nocca, false);
107  r.bpot= zero_functions_compressed<double_complex,3>(world, noccb, false);
108  }
109  for (auto&& vv : v) {
110  compress(world, vv.afunction,false);
111  compress(world, vv.bfunction,false);
112  compress(world, vv.apot,false);
113  compress(world, vv.bpot,false);
114  }
115  world.gop.fence();
116 
117  for (int i=0; i<m; ++i) {
118  for (int j=0; j<n; ++j) {
119  gaxpy(world,double_complex(1.0),result[i].afunction,c(j,i),v[j].afunction,false);
120  gaxpy(world,double_complex(1.0),result[i].bfunction,c(j,i),v[j].bfunction,false);
121  if (v[j].apot.size()>0) gaxpy(world,double_complex(1.0),result[i].apot,c(j,i),v[j].apot,false);
122  if (v[j].bpot.size()>0) gaxpy(world,double_complex(1.0),result[i].bpot,c(j,i),v[j].bpot,false);
123  }
124  }
125 
126  if (fence) world.gop.fence();
127  return result;
128  }
