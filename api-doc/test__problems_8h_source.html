34 /** \file test_problems.h
35  \brief Provides test problems for examining the convergence of
36  embedded (Dirichlet) boundary conditions.
37 
38  The auxiliary PDE being solved is
39  \f[ \nabla^2 u - p(\varepsilon) S (u-g) = \varphi f, \f]
40  where
41  - \f$u\f$ is the solution function
42  - \f$\varepsilon\f$ is the thickness of the boundary layer
43  - \f$p(\varepsilon)\f$ is the penalty prefactor, \f$2/\varepsilon\f$
44  seems to work well.
45  - \f$S\f$ is the surface function
46  - \f$g\f$ is the Dirichlet condition to be enforced on the surface
47  - \f$\varphi\f$ is the domain mask (1 inside, 0 outside, blurry on the
48  border)
49  - \f$f\f$ is the inhomogeneity.
50 
51  The available test problems are
52  -# A sphere of radius \f$R\f$ with \f$g = Y_0^0\f$, homogeneous
53  (ConstantSphere)
54  -# A sphere of radius \f$R\f$ with \f$g = Y_1^0\f$, homogeneous
55  (CosineSphere)
56  -# A sphere of radius \f$R\f$ with \f$g = Y_2^0\f$, homogeneous
57  (Y20Sphere)
58  -# A sphere of radius \f$R\f$ with \f$g = Y_0^0\f$, inhomogeneous
59  \f$ f = 1 \f$ (InhomoConstantSphere)
60 
61  This file sets up the various details of the problems... the main program
62  is found in embedded_dirichlet.cc. */

102 /** \brief Abstract base class for embedded Dirichlet problems. */
103 class EmbeddedDirichlet : public FunctionFunctorInterface<double, 3> {
104  private:
105  EmbeddedDirichlet() {}
106  int initial_level;
107  int k;
108  double thresh;
109  std::string penalty_name;
110 
111  protected:
112  DomainMaskInterface *dmi;
113  SignedDFInterface<3> *sdfi;
114  double penalty_prefact, eps;
115  std::string problem_name;
116  std::string problem_specific_info;
117  std::string domain_mask_name;
118 
119  // is the problem homogeneous?
120  virtual bool isHomogeneous() const = 0;
121 
122  public:
123  /// which function to use when projecting:
124  /// -# the weighted surface (SURFACE)
125  /// -# the rhs of the auxiliary DE (DIRICHLET_RHS)
126  /// -# the exact solution (EXACT)
127  /// -# the domain mask (DOMAIN_MASK)
128  FunctorOutput fop;
129 
130  /// \brief Sets up the data for the problem-inspecific parts.
131  ///
132  /// Also sets the FunctionDefaults for the appropriate dimension.
133  EmbeddedDirichlet(double penalty_prefact, std::string penalty_name,
134  double eps, int k, double thresh, Mask mask)
135  : k(k), thresh(thresh), penalty_name(penalty_name), dmi(nullptr),
136  sdfi(nullptr), penalty_prefact(penalty_prefact), eps(eps),
137  fop(DIRICHLET_RHS) {
138 
139  // calculate some nice initial projection level
140  // should be no lower than 6, but may need to be higher for small
141  // eps
142  initial_level = ceil(log(4.0 / eps) / log(2.0) - 4);
143  if(initial_level < 6)
144  initial_level = 6;
145  FunctionDefaults<3>::set_initial_level(initial_level);
146 
147  switch(mask) {
148  case LLRV:
149  domain_mask_name = "LLRV";
150  dmi = new LLRVDomainMask(eps);
151  break;
152  case Gaussian:
153  domain_mask_name = "Gaussian";
154  dmi = new GaussianDomainMask(eps);
155  break;
156  default:
157  error("Unknown mask");
158  break;
159  }
160  }
161 
162  virtual ~EmbeddedDirichlet() {
163  if(sdfi != nullptr)
164  delete sdfi;
165  if(dmi != nullptr)
166  delete dmi;
167  }
168 
169  /// \brief Load balances using the provided Function
170  void load_balance(World &world, const Function<double, 3> &f)
171  const {
172 
173  LoadBalanceDeux<3> lb(world);
174  lb.add_tree(f, DirichletLBCost<3>(1.0, 1.0));
175  // set this map as the default
176  FunctionDefaults<3>::redistribute(world, lb.load_balance(2.0,
177  false));
178  }
179 
180  /// \brief Do a standard Printout of the problem details
181  void printout() const {
182  printf("Solving problem: %s\nWavelet Order: %d\nThreshold: %.6e" \
183  "\nEpsilon: %.6e\nPenalty Prefactor, %s: %.6e\nUsing %s " \
184  "domain masking\n%s\n",
185  problem_name.c_str(), k, thresh, eps, penalty_name.c_str(),
186  penalty_prefact, domain_mask_name.c_str(),
187  problem_specific_info.c_str());
188  fflush(stdout);
189  }
190 
191  /// \brief The operator for projecting a MADNESS function.
192  double operator() (const Vector<double, 3> &x) const {
193  switch(fop) {
194  case EXACT:
195  return ExactSol(x);
196  break;
197  case DIRICHLET_RHS:
198  if(isHomogeneous())
199  return -DirichletCond(x) * dmi->surface(sdfi->sdf(x)) *
200  penalty_prefact;
201  else
202  return dmi->mask(sdfi->sdf(x)) * Inhomogeneity(x) -
203  DirichletCond(x) * dmi->surface(sdfi->sdf(x)) *
204  penalty_prefact;
205  break;
206  case SURFACE:
207  return dmi->surface(sdfi->sdf(x)) * penalty_prefact;
208  break;
209  case DOMAIN_MASK:
210  return dmi->mask(sdfi->sdf(x));
211  break;
212  default:
213  error("shouldn't be here...");
214  return 0.0;
215  break;
216  }
217  }
218

222 
223  virtual double Inhomogeneity(const Vector<double, 3> &x) const = 0;
224 
225  /// \brief The surface area of the domain

227 
228  /// \brief The volume of the domain

230

232  /// solution to the exact solution
233  virtual std::vector< Vector<double, 3> > check_pts() const {
234  return std::vector< Vector<double, 3> >();
235  }
