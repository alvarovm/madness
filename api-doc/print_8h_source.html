106 /// Easy printing of lists.

108 /// \tparam T Type stored in the list.
109 /// \param[in,out] s The output stream.
110 /// \param[in] c The list.
111 /// \return The output stream (for chaining).
112 template <typename T>
113 std::ostream &operator<<(std::ostream &s, const std::list<T> &c) {
114  s << "[";

146 /// Easy printing of fixed dimension arrays.
147 
148 /// STL I/O already does char (thus the \c enable_if business).
149 /// \tparam T Type of data in the array.
150 /// \tparam N Size of the array.
151 /// \param[in,out] s The output stream.
152 /// \param[in] v The array.
153 /// \return The output stream (for chaining).
154 template <typename T, std::size_t N>
155 typename std::enable_if<!std::is_same<T, char>::value, std::ostream &>::type
156 operator<<(std::ostream &s, const T (&v)[N]) {
157  s << "[";
158  for (std::size_t i = 0; i < N; ++i) {
159  s << v[i];
160  if (i != (N - 1))
161  s << ",";
162  }
163  s << "]";
164  return s;
165 }

178  /// Helper function for \c print. Base case.
179 
180  /// This gets called recursively when there are no items left to print.
181  /// \param[in,out] out Output stream.
182  /// \return The output stream (for chaining).
183  inline std::ostream& print_helper(std::ostream& out) {
184  return out;
185  }
186 
187  /// \brief Helper function for \c print. Prints the first item (\c t) and
188  /// recursively passes on the other items.
189 
190  /// \tparam T Type of the item to print in this call.
191  /// \tparam Ts Argument pack type for the remaining items.
192  /// \param[in,out] out Output stream.
193  /// \param[in] t The item to print in this call.
194  /// \param[in] ts The remaining items in the argument pack (they get
195  /// recursively passed on).
196  /// \return The output stream (for chaining).
197  template <typename T, typename... Ts>
198  inline std::ostream& print_helper(std::ostream& out,
199  const T& t, const Ts&... ts) {
200  using madness::operators::operator<<;
201  out << ' ' << t;
202  return print_helper(out, ts...);
203  }
204 
205  /// \brief Print items to \c std::cout (items separated by spaces) and
206  /// terminate with a new line
207 
208  /// The first item is printed here so that it isn't preceded by a space.
209  /// \tparam T Type of the first item to be printed.
210  /// \tparam Ts Argument pack type for the items to be printed.
211  /// \param[in] t The first item to be printed.
212  /// \param[in] ts The remaining items to be printed in the argument pack.
213  template<typename T, typename... Ts>
214  void print(const T& t, const Ts&... ts) {
215  using madness::operators::operator<<;
216  ScopedMutex<Mutex> safe(detail::printmutex);
217  std::cout << t;
218  print_helper(std::cout, ts...) << ENDL;
219  }
220 
221  /// \brief Print items to \c std::cerr (items separated by spaces) and
222  /// terminate with a new line
223 
224  /// The first item is printed here so that it isn't preceded by a space.
225  /// \tparam T Type of the first item to be printed.
226  /// \tparam Ts Argument pack type for the items to be printed.
227  /// \param[in] t The first item to be printed.
228  /// \param[in] ts The remaining items to be printed in the argument pack.
229  template<typename T, typename... Ts>
230  void print_error(const T& t, const Ts&... ts) {
231  using madness::operators::operator<<;
