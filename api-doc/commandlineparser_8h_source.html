23  // parse command line arguments
24  // mp2 --mp2='maxiter 10; freeze 1' --dft:maxiter=20 --Xmpi:debug=true
25  commandlineparser(int argc, char **argv) {
26  set_defaults();
27  std::vector<std::string> allArgs_raw(argv, argv + argc);
28  for (auto &a : allArgs_raw) {
29  // special treatment for the input file: no hyphens
30  a=check_for_input_file(a);
31  a= remove_first_equal(remove_front_hyphens(a));
32  std::replace_copy(a.begin(), a.end(), a.begin(), '=', ' ');
33  std::string key, val;
34  std::stringstream sa(a);
35  sa >> key;
36  val=a.substr(key.size());
37  set_keyval(key,val);
38  }
39  }

84  static std::vector<std::string> split(std::string s, const std::string delimiter) {
85  std::size_t pos = 0;

87  std::vector<std::string> result;
