241  void set_derived_values(const Molecule& molecule, const AtomicBasisSet& aobasis, const commandlineparser& parser) {
242  std::string inputfile=parser.value("input");
243  std::string prefix=commandlineparser::remove_extension(commandlineparser::base_name(inputfile));
244  if (prefix!="input") set_derived_value("prefix",prefix);
245 
246  for (size_t iatom = 0; iatom < molecule.natom(); iatom++) {
247  if (molecule.get_pseudo_atom(iatom)){
248  set_derived_value("pure_ae",false);
249  continue;
250  }
251  }
252  set_derived_value("aobasis",molecule.guess_file());;
253  const int n_core = molecule.n_core_orb_all();
254 
255 
256  std::vector<double> proto=get<std::vector<double> >("protocol");
257  // No ... The accuracy of computation is INDEPENDENT of the convergence requirement
258  // --- actually need more precision than convergence threshold in order to have
259  // variational principle working and for robust convergence
260  //proto.back()=get<double>("econv");
261  set_derived_value("protocol",proto);
262  // No ... the energy is variational! Don't need more accuracy in dconv --- in fact the opposite is true.
263  // set_derived_value("dconv",sqrt(get<double>("econv"))*0.1);
264 
265  double z = molecule.total_nuclear_charge();
266  const double charge=get<double>("charge");
267  int nelec = int(z - charge - n_core*2);
268  if (fabs(nelec+charge+n_core*2-z) > 1e-6) {
269  error("non-integer number of electrons?", nelec+charge+n_core*2-z);
270  }
271 
272  set_derived_value("nalpha",(nelec + nopen())/2);
273  set_derived_value("nbeta",(nelec - nopen())/2);
274 
275  if (nalpha() < 0) error("negative number of alpha electrons?", nalpha());
276  if (nbeta() < 0) error("negative number of beta electrons?", nbeta());
277  if ((nalpha()+nbeta()) != nelec) error("nalpha+nbeta != nelec", nalpha()+nbeta());
278  if (nalpha() != nbeta()) set_derived_value("spin_restricted",false);
279 
280  set_derived_value("nmo_alpha",nalpha() + nvalpha());
281  set_derived_value("nmo_beta",nbeta() + nvbeta());
282 
283  // Ensure we have enough basis functions to guess the requested
284  // number of states ... a minimal basis for a closed-shell atom
285  // might not have any functions for virtuals.
286  int nbf = aobasis.nbf(molecule);
287  if ((nmo_alpha()>nbf) or (nmo_beta()>nbf)) error("too few basis functions?", nbf);
288 // nmo_alpha = std::min(nbf,nmo_alpha);
289 // nmo_beta = std::min(nbf,nmo_beta);
290 // if (nalpha>nbf || nbeta>nbf) error("too few basis functions?", nbf);
291 // nvalpha = nmo_alpha - nalpha;
292 // nvbeta = nmo_beta - nbeta;
293 
294  // Unless overridden by the user use a cell big enough to
295  // have exp(-sqrt(2*I)*r) decay to 1e-6 with I=1ev=0.037Eh
296  // --> need 50 a.u. either side of the molecule
297  set_derived_value("l",molecule.bounding_cube() + 50.0);
298 
299  set_derived_value("lo",molecule.smallest_length_scale());
300 
301  // set highest possible point group for symmetry
302  if (do_localize()) set_derived_value("pointgroup",std::string("c1"));
303  else set_derived_value("pointgroup",molecule.get_pointgroup());
304 
305  // above two lines will not override user input, so check input is sane
306  if (do_localize() and do_symmetry()) {
307  error("\n\nsymmetry and localization cannot be used at the same time\n"
308  "switch from local to canonical orbitals (keyword canon)\n\n");
309  }
310 
311  //NWChem interface doesn't support geometry optimization
312  if (get<bool>("gopt") && nwfile() != "none") error("NWchem initialization only supports single point energy calculations.");
313 
314  //NWChem only supports Boys localization (or canonical)
315  if (nwfile() != "none") {
316  set_derived_value("localize",std::string("boys"));
317  //Error if user requested something other than Boys
318  if(localize_method() != "boys" and localize_method() != "canon") error("NWchem initialization only supports Boys localization");
319  }
320  }
