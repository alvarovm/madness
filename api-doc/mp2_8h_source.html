106  double value(const Tensor<double>& x) {
107 
108  // fast return if the reference is already solved at this geometry
109  double xsq = x.sumsq();
110  if (xsq == coords_sum) return nemo_ptr->get_calc()->current_energy;
111 
112  nemo_ptr->get_calc()->molecule.set_all_coords(x.reshape(nemo_ptr->get_calc()->molecule.natom(), 3));
113  coords_sum = xsq;
114  nemo_ptr->value(x);
115 
116  MolecularOrbitals<double,3> mos(nemo_ptr->get_calc()->amo,nemo_ptr->get_calc()->aeps);
117  reset_orbitals(mos);
