21 {
22  register int32_t Val;

29 __INLINE__ int StoreConditionalSigned32( volatile int32_t *pVar, int32_t Val )
30 {
31  register int rc = 1; // assume success
32  asm volatile (" stwcx. %2,0,%1;"
33  " beq 1f;" // conditional store succeeded
34  " li %0,0;"
35  "1:;"
36  : "=b" (rc)
37  : "b" (pVar),
38  "b" (Val),
39  "0" (rc)
40  : "cc", "memory" );
41  return(rc);
42 }
