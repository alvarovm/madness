108  /// normalize the nemos
109  template<typename T, std::size_t NDIM>
110  void static normalize(std::vector<Function<T,NDIM> >& nemo,
111  const Function<double,NDIM> metric=Function<double,NDIM>()) {
112 
113  if (nemo.size()==0) return;
114  World& world=nemo[0].world();
115  // compute the norm of the reconstructed orbitals, includes the factor
116  std::vector<Function<T,NDIM> > mos = (metric.is_initialized()) ? metric*nemo : nemo;
117  std::vector<double> norms = norm2s(world, mos);
118 
119  // scale the nemos, excludes the nuclear correlation factor
120  std::vector<double> invnorm(norms.size());
121  for (std::size_t i = 0; i < norms.size(); ++i)
122  invnorm[i] = 1.0 / norms[i];
123  scale(world, nemo, invnorm);
124  }

126  template<typename T>
127  static Tensor<T> Q2(const Tensor<T>& s) {
128  Tensor<T> Q = -0.5*s;
129  for (int i=0; i<s.dim(0); ++i) Q(i,i) += 1.5;
130  return Q;

133  /// orthonormalize the vectors
134  template<typename T, std::size_t NDIM>
135  void orthonormalize(std::vector<Function<T,NDIM> >& nemo,
136  const Function<double,NDIM> metric=Function<double,NDIM>(),
137  const double trantol=FunctionDefaults<NDIM>::get_thresh()*0.01) const {
138 
139  if (nemo.size()==0) return;
140  normalize(nemo,metric);
141  double maxq;
142  do {
143  std::vector<Function<T,NDIM> > Rnemo = (metric.is_initialized()) ? metric*nemo : nemo;
144  Tensor<T> Q = Q2(matrix_inner(world, Rnemo, Rnemo));
145  maxq=0.0;
146  for (int i=0; i<Q.dim(0); ++i)
147  for (int j=0; j<i; ++j)
148  maxq = std::max(maxq,std::abs(Q(i,j)));
149 
150  Q.screen(trantol); // ???? Is this really needed?
151  nemo = transform(world, nemo, Q, trantol, true);
152  truncate(world, nemo);
153 // if (world.rank() == 0) print("ORTHOG2: maxq trantol", maxq, trantol);
154 
155  } while (maxq>0.01);
156  normalize(nemo,metric);
157  }

199  /// compute kinetic energy as square of the "analytical" expectation value
200 
201  /// @param[in] the nemo orbitals F
202  /// @return T = 1/2 \sum_i \int R^2 U1.U1 F^2 + 2 R^2 U1.grad(F) + R^2 grad(F)^2
203  template<typename T, std::size_t NDIM>
204  double compute_kinetic_energy(const std::vector<Function<T,NDIM> >& nemo) const {
205 
206  // T = 0.5\sum_i \int R^2 U1.U1 F^2 - 2 R^2 U1.grad(F) F + R^2 grad(F)^2
207  // = 0.5 (<U1.U1 | rho > + <R^2|grad(F)^2> - 2<R^2 | U1.grad(F) >)
208  // note: U1=-grad(R)/R
209  auto id=nemo.front().world().id();
210  auto id1=R_square.world().id();
211  auto worldid=world.id();
212  world.gop.fence();
213  real_function_3d dens=dot(world,nemo,nemo)*R_square;
214  real_function_3d U1dotU1=real_factory_3d(world)
215  .functor(NuclearCorrelationFactor::U1_dot_U1_functor(ncf.get()));
216  double ke1=inner(dens,U1dotU1);
217 
218  double ke2=0.0;
219  double ke3=0.0;
220  double ke3_real=0.0;
221  double ke3_imag=0.0;
222 
223  for (int axis = 0; axis < NDIM; axis++) {
224  real_derivative_3d D = free_space_derivative<double, NDIM>(world, axis);
225  const std::vector<Function<T,NDIM> > dnemo = apply(world, D, nemo);
226 
227  real_function_3d term2=dot(world,dnemo,nemo)*ncf->U1(axis);
228  double tmp=-2.0*inner(R_square,term2);
229 // ke3_real -=2.0*std::real(tmp);
230 // ke3_imag -=2.0*std::imag(tmp);
231  ke3 +=tmp;
232 
233  const real_function_3d term1=dot(world,dnemo,dnemo);
234  world.gop.fence();
235  ke2 += inner(term1,R_square);
236 
237  }
238 // if (ke3_imag>1.e-8) {
239 // print("kinetic energy, imaginary part: ",ke3_imag);
240 // MADNESS_EXCEPTION("imaginary kinetic energy",1);
241 // }
242 // double ke=2.0*(ke1+ke2+ke3_real); // closed shell
243  double ke=2.0*(ke1+ke2+ke3); // closed shell
244  return 0.5*ke;
245  }

247  /// compute kinetic energy as square of the "analytical" derivative of the orbitals
248 
249  /// @param[in] the nemo orbitals F
250  /// @return T = 1/2 \sum_i || grad(R)*F_i + R*grad(F_i)||^2
251  template<typename T, std::size_t NDIM>
252  double compute_kinetic_energy1(const std::vector<Function<T,NDIM> >& nemo) const {
253  timer timer1(world);
254  double ke=0.0;
255  for (int i=0; i<nemo.size(); ++i) {
256  double fnorm2=norm2(world,-1.0*R*ncf->U1vec()*nemo[i] + R*grad(nemo[i]));
257  ke+=2.0*fnorm2*fnorm2;
258  }
259  timer1.end("compute_kinetic_energy1");
260  return 0.5*ke;
261  }

263 
264  /// compute kinetic energy as square of the "analytical" derivative of the orbitals
265 
266  /// @param[in] the nemo orbitals F
267  /// @return T = 1/2 \sum_i || grad(R)*F_i + R*grad(F_i)||^2
268  template<typename T, std::size_t NDIM>
269  double compute_kinetic_energy1a(const std::vector<Function<T,NDIM> >& nemo) const {
270  timer timer1(world);
271  double ke=0.0;
272  for (int i=0; i<NDIM; ++i) {
273  std::vector< std::shared_ptr< Derivative<T,NDIM> > > grad=
274  gradient_operator<T,NDIM>(world);
275  double fnorm2=norm2(world,R*(-1.0*ncf->U1(i)*nemo + apply(world,*(grad[i]),nemo)));
276  ke+=2.0*fnorm2*fnorm2;
277  }
278  timer1.end("compute_kinetic_energy1a");
279  return 0.5*ke;
280  }
281 
282  /// compute kinetic energy as direct derivative of the orbitals (probably imprecise)
283 
284  /// @param[in] the nemo orbitals F
285  /// @return T = 1/2 \sum_i || grad(R*F_i)||^2
286  template<typename T, std::size_t NDIM>
287  double compute_kinetic_energy2(const std::vector<Function<T,NDIM> >& nemo) const {
288 
289  // it's ok to use phi here, no regularization necessary for this eigenvalue
290  double E_kin = 0.0;
291  for (int axis = 0; axis < 3; axis++) {
292  real_derivative_3d D = free_space_derivative<double, 3>(world, axis);
293  const vecfuncT dphi = apply(world, D,R*nemo);
294  E_kin += 0.5 * (inner(world, dphi, dphi)).sum();
295  // -1/2 sum <Psi|Nabla^2|Psi> = 1/2 sum <NablaPsi|NablaPsi> (integration by parts)
296  }
297  E_kin *= 2.0; // 2 because closed shell
298  return E_kin;
299  }

302  bool check_convergence(const std::vector<double> energies,
303  const std::vector<double> oldenergies, const double bsh_norm,
304  const double delta_density, const CalculationParameters& param,
305  const double econv, const double dconv) const {
306 
307  double maxenergychange=fabs(energies.size()-oldenergies.size()); // >0 if oldenergyvec not initialized
308  for (auto iter1=energies.begin(), iter2=oldenergies.begin();
309  (iter1!=energies.end() and iter2!=oldenergies.end()); iter1++, iter2++) {
310  maxenergychange=std::max(maxenergychange,fabs(*iter1 - *iter2));
311  }
312  double delta_energy=fabs(energies[0]-oldenergies[0]);
313 
314  bool bsh_conv=param.converge_bsh_residual() ? bsh_norm<dconv : true;
315  bool total_energy_conv=param.converge_total_energy() ? delta_energy<econv : true;
316  bool each_energy_conv=param.converge_each_energy() ? maxenergychange<econv*3.0 : true;
317  bool density_conv=param.converge_density() ? delta_density<dconv : true;
318 
319  if (world.rank()==0 and param.print_level()>=2) {
320  std::stringstream line;
321  line << "convergence: bshresidual, energy change, max energy change, density change "
322  << std::scientific << std::setprecision(1)
323  << bsh_norm << " " << delta_energy << " "
324  << maxenergychange << " " << delta_density;
325  print(line.str());
326  }
327 
328  return (bsh_conv and density_conv and each_energy_conv and total_energy_conv);

539  /// the Laplacian of the density
540

542  /// with a Green's function (which is reasonably stable), but not on its own!
543  ///
544  /// The Laplacian of the cuspy density is numerically fairly unstable:
545  /// - a singular term may be rewritten using the nuclear potential (see below)
546  /// - the Laplacian of the regularized density is still very noisy
547  ///
548  /// It may be computed as
549  /// \f[
550  /// \Delta \rho = \Delta (R^2 \rho_R)
551  /// = \Delta (R^2) \rho_R + 2\nabla R \nabla \rho_R + R^2 \Delta \rho_R
552  /// = 2 R^2 U1^2 \rho_R -4 R^2 ( U-V ) \rho_R + R^2 \Delta\rho_R

558  /// \f]
