21 struct divide_add_interpolate {
22

24  double thresh_low=1.e-7;

28 
29  divide_add_interpolate(double hi, double lo, double eps_regularize) :
30  thresh_high(hi), thresh_low(lo), eps_regularize(eps_regularize),
31  log_high(log10(thresh_high)), log_low(log10(thresh_low)) {}
32 
33  std::size_t get_result_size() const {
34  return 1;
35  }

37  /// @param[in] t vector containing: oep, refdensity, longrange
38  /// @return (num1/denom1 - num2/denom2) * mask + (1-mask)*longrange
39  std::vector<Tensor<double> > operator()(const Key<3> & key,
40  const std::vector<Tensor<double> >& t) const {
41 
42  const Tensor<double>& refdens=t[0];
43  const Tensor<double>& num1=t[1];
44  const Tensor<double>& denom1=t[2];
45  const Tensor<double>& num2=t[3];
46  const Tensor<double>& denom2=t[4];
47  const Tensor<double>& longrange=t[5];
48 
49  // set 1 if above munge interval (dens > thresh_high) or 0 if below munge interval (dens < thresh_low)
50  // interpolate inbetween with logarithmic form of (r - lo)/(hi - lo) because this yields near linear behavior
51  Tensor<double> U = copy(refdens); // copy refdens in oder to have the same dimension in the tensor
52  U.fill(1.0); // start with a fuction that is 1.0 everywhere
53  ITERATOR(
54  U,
55  double r = refdens(IND);
56  double result=num1(IND)/(denom1(IND)+eps_regularize)
57  - num2(IND)/(denom2(IND)+eps_regularize);
58  if (square_denominator) result=num1(IND)/(denom1(IND)*denom1(IND)+eps_regularize)
59  - num2(IND)/(denom2(IND)*denom2(IND)+eps_regularize);
60  if (r > thresh_high) {
61  U(IND) = result;
62  } else if (r < thresh_low) {
63  U(IND) = longrange(IND);
64  } else {
65  // mask = 1 if refdens>hi, 0 if refdens < lo, interpolate in between
66  double mask=(log10(refdens(IND)) - log_low)/(log_high - log_low);
67  U(IND)=mask*result + (1.0-mask)*longrange(IND);
68  }
69  );
70  return std::vector<Tensor<double> > (1,U);
71  }
72 };

284  /// compute Slater potential (Kohut, 2014, equation (15))
285  real_function_3d compute_slater_potential(const vecfuncT& nemo) const {
286 
287  Exchange<double,3> K;
288  K.set_parameters(R_square*nemo,nemo,reference->get_calc()->param.lo());
289  const vecfuncT Knemo = K(nemo);
290  // 2.0*R_square in numerator and density (rho) cancel out upon division
291  real_function_3d numerator = -1.0*dot(world, nemo, Knemo);
292  real_function_3d rho = dot(world, nemo, nemo);
293 
294  // long-range asymptotic behavior for Slater potential is \int 1/|r-r'| * |phi_HOMO|^2 dr'
295  // in order to compute this lra, use Coulomb potential with only HOMO density (= |phi_HOMO|^2)
296 // Coulomb J(world);
297 // J.reset_poisson_operator_ptr(param.lo(),param.econv());
298 // real_function_3d lra = -1.0*J.compute_potential(R_square*square(nemo[homo_ind]));
299  Coulomb<double,3> J(world,this);
300  real_function_3d lra=-1.0/(param.nalpha()+param.nbeta())*J.compute_potential(this);
301 // print("compute long-range part of the Slater potential from the full molecular density");
302  if (oep_param.saving_amount() >= 3) save(lra, "lra_slater");

305  real_function_3d one=real_factory_3d(world).functor([](const coord_3d& r){return 1.0;});

316  /// compute the total kinetic energy density of equation (6) from Kohut
317 
318  /// return without the NCF and factor 2 for closed shell !
319  real_function_3d compute_total_kinetic_density(const vecfuncT& nemo) const {
320 
321  // compute tau = 1/2 * sum |\nabla phi_i|^2
322  // = 1/2 * sum {(\nabla R)^2 * nemo_i^2 + 2 * R * nemo_i * (\nabla R) * (\nabla nemo_i)) + R^2 * (\nabla nemo_i)^2}
323  // = 1/2 * R^2 * sum {U1dot * nemo_i^2 + 2 * nemo_i * U1 * (\nabla nemo_i)) + (\nabla nemo_i)^2}
324 
325  // get \nabla R and (\nabla R)^2 via and U1 = -1/R \nabla R and U1dot = (1/R \nabla R)^2
326  const vecfuncT U1 = this->ncf->U1vec();
327  NuclearCorrelationFactor::U1_dot_U1_functor u1_dot_u1(ncf.get());
328  const real_function_3d U1dot = real_factory_3d(world).functor(u1_dot_u1).truncate_on_project();
329 
330  real_function_3d u1u1term=U1dot*dot(world,nemo,nemo);
331  real_function_3d u1dnterm=real_factory_3d(world).compressed();
332  real_function_3d dndnterm=real_factory_3d(world).compressed();
333 
334  for (int idim=0; idim<3; ++idim) {
335  real_derivative_3d D(world,idim);
336  if(param.dft_deriv() == "bspline") D.set_bspline1();
337  vecfuncT nemo_copy=copy(world,nemo);
338  refine(world,nemo_copy);
339  std::vector<real_function_3d> dnemo=apply(world,D,nemo_copy);
340  u1dnterm+=U1[idim]*dot(world,nemo_copy,dnemo);
341  refine(world,dnemo);
342  dndnterm+=dot(world,dnemo,dnemo);
343  }
344 
345  real_function_3d tau=0.5*(u1u1term - 2.0*u1dnterm + dndnterm);
346 
347  return tau;
348 
349  }

351  /// compute the Pauli kinetic energy density divided by the density tau_P/rho with equation (16) from Ospadov, 2017
352  real_function_3d compute_Pauli_kinetic_density(const vecfuncT& nemo) const {
353 
354  // compute the numerator tau_P
355 
356  // get \nabla nemo
357  std::vector<vecfuncT> grad_nemo(nemo.size());
358  for (long i = 0; i < nemo.size(); i++) {
359  vecfuncT nemo_copy=copy(world,nemo);
360  refine(world,nemo_copy);
361 
362  if(param.dft_deriv() == "bspline") grad_nemo[i] = grad_bspline_one(nemo_copy[i]); // gradient using b-spline
363  else grad_nemo[i] = grad(nemo_copy[i]); // default gradient using abgv
364  }
365 
366  vecfuncT grad_nemo_term;
367  for (size_t i = 0; i < nemo.size(); i++) {
368  for (size_t j = i + 1; j < nemo.size(); j++) {
369  vecfuncT tmp = nemo[i]*grad_nemo[j] - nemo[j]*grad_nemo[i];
370  grad_nemo_term.push_back(dot(world, tmp, tmp));
371  }
372  }
373  real_function_3d numerator = sum(world, grad_nemo_term); // numerator = tau_P * 2 * rho / R^4
374 
375  return numerator;

394  }

528  double compute_exchange_energy_vir(const vecfuncT& nemo, const real_function_3d Vx) const {
529

531  auto monomial_x = [] (const coord_3d& r) {return r[0];};

533  auto monomial_z = [] (const coord_3d& r) {return r[2];};
